reuse content in templates

It is possible to embed content in a template. All instances of the template inherit this content. The content embedded in the template is merged with the content of the page.


That's useful for a global navigation for example. In this example the navbar is embedded and the selected item is defined by the property selectedNavItem. The template navbar has a single variable that is not defined by its own form and needs to be provided via a parameter selectedNavItem.

Change selectedNavItem to Tutorial to see the effect.

  • templates
{
 "selectedNavItem": "rich text",
 "name": "reuse content in templates",
 "index": 3,
 "template": "/embed-navbar-example.json",
 "path": "/tutorial/advanced templating/reuse content in templates.json",
 "parent": "/tutorial/advanced templating",
 "navbar": {
  "brand": "tutorial",
  "navItems": [
   {
    "_c4a_path": "navItems.0",
    "label": "rich text",
    "page": "/tutorial/rich media/rich text.json"
   },
   {
    "_c4a_path": "navItems.1",
    "label": "images",
    "page": "/tutorial/rich media/images.json"
   }
  ],
  "template": "/navbar.json",
  "name": "navbar",
  "brand-link": "/tutorial/index.json",
  "path": "/tutorial/components/navbar",
  "parent": "/tutorial/components"
 }
}
{{#navbar}}
   {{>navbar selectedNavItem=../selectedNavItem}}
{{/navbar}}
navbar 
<nav class="navbar navbar-default">
 <div class="container-fluid">
 <!-- Brand and toggle get grouped for better mobile display -->
 <div class="navbar-header">
 <button type="button" class="navbar-toggle collapsed" data-toggle="collapse" data-target="#bs-example-navbar-collapse-1">
 <span class="sr-only">Toggle navigation</span>
 <span class="icon-bar"></span>
 <span class="icon-bar"></span>
 <span class="icon-bar"></span>
 </button>
 <a class="navbar-brand" href="{{link-path brand-link}}">{{brand}}</a>
 </div>

 <!-- Collect the nav links, forms, and other content for toggling -->
 <div class="collapse navbar-collapse" id="bs-example-navbar-collapse-1">
 <ul class="nav navbar-nav">
 {{#navItems}} 
 <li {{#equals ../selectedNavItem label}}class="active"{{/equals}}><a href="{{link page}}">{{label}}</a></li>
 {{/navItems}}
 </div><!-- /.navbar-collapse -->
 </div><!-- /.container-fluid -->
</nav>
